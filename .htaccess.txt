# Je protège uniquement les fichiers sensibles hors du dossier public
<FilesMatch "\.(sql|log|env)$">
    Order Allow,Deny
    Deny from all
</FilesMatch>

# J'autorise uniquement l'accès au dossier public
<IfModule mod_rewrite.c>
    RewriteEngine On
    
    # Je redirige tout vers le dossier public
    RewriteCond %{REQUEST_URI} !^/public/
    RewriteCond %{REQUEST_FILENAME} !-f
    RewriteCond %{REQUEST_FILENAME} !-d
    RewriteRule ^(.*)$ /public/$1 [L]
</IfModule>
